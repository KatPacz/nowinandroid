@AndroidEntryPoint
class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent {
            YourAppTheme { // Our custom theme from Phase 1
                val navController = rememberNavController()
                NavHost(navController = navController, startDestination = Routes.HOME) {
                    composable(Routes.HOME) {
                        HomeScreen(
                            onDeckSelected = { deckId ->
                                navController.navigate("pre_game/$deckId")
                            }
                        )
                    }
                    composable(Routes.PRE_GAME) { backStackEntry ->
                        val deckId = backStackEntry.arguments?.getString("deckId")?.toIntOrNull()
                        if (deckId != null) {
                            PreGameScreen(
                                deckId = deckId,
                                onStartGame = { /* Navigate to gameplay screen */ }
                            )
                        }
                    }
                    // Add other composables for gameplay, settings, etc. here
                }
            }
        }
    }
}
