package com.google.samples.apps.partygameapp.data.database
import androidx.room.Database

@Database(entities = [CategoryEntity::class, DeckEntity::class, CardEntity::class], version = 1)
    abstract class AppDatabase : RoomDatabase() {
        abstract fun deckDao(): DeckDao
        // We'll add other DAOs here later (PlayerDao, CategoryDao)
    }
abstract class AppDatabase : RoomDatabase() {
    // DAOs...

    class PrepopulateCallback(
        private val context: Context,
        private val provider: Provider<CategoryDao>,
        private val providerDeck: Provider<DeckDao>,
        private val providerCard: Provider<CardDao>
    ) : RoomDatabase.Callback() {
        // Using CoroutineScope to perform this work off the main thread
        private val applicationScope = CoroutineScope(SupervisorJob())

        override fun onCreate(db: SupportSQLiteDatabase) {
            super.onCreate(db)
            applicationScope.launch(Dispatchers.IO) {
                prepopulateDatabase()
            }
        }

        suspend fun prepopulateDatabase() {
            // Define default categories
            val categories = mapOf(
                "Animals" to "#6B8E85", // Muted Green
                "Food" to "#F4A261", // Peach
                "Movies" to "#4A5859" // Slate Gray
            )

            categories.forEach { (categoryName, color) ->
                val categoryId = provider.get().insertCategory(CategoryEntity(name = categoryName, colorHex = color))

                // Read the corresponding text file from assets
                val deckFileName = "${categoryName.lowercase()}.txt"
                val deckName = categoryName
                val deckId = providerDeck.get().insertDeck(DeckEntity(name = deckName, description = "Default deck", categoryId = categoryId.toInt()))

                try {
                    context.assets.open("decks/en/$deckFileName").bufferedReader().useLines { lines ->
                        lines.forEach { line ->
                            if (line.isNotBlank()) {
                                providerCard.get().insertCard(CardEntity(text = line, deckId = deckId.toInt()))
                            }
                        }
                    }
                } catch (e: IOException) {
                    Log.e("PrepopulateDB", "Error reading asset file: $deckFileName", e)
                }
            }
        }
    }
}
