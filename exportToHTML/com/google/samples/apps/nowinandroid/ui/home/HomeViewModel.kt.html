package com.google.samples.apps.partygameapp.ui.home

import dagger.hilt.android.lifecycle.HiltViewModel
import jakarta.inject.Inject

@HiltViewModel
class HomeViewModel @Inject constructor(
    private val deckRepository: DeckRepository
) : ViewModel() {

    // Using StateFlow to hold the UI state
    private val _uiState = MutableStateFlow<HomeScreenState>(HomeScreenState.Loading)
    val uiState: StateFlow<HomeScreenState> = _uiState.asStateFlow()

    init {
        viewModelScope.launch {
            // For now, let's add some mock data if the DB is empty
            // In the next phase, we'll load this from assets
            deckRepository.getAllDecks().collect { decks ->
                _uiState.value = HomeScreenState.Success(decks)
            }
        }
    }
}

// Defines the possible states of our screen
sealed interface HomeScreenState {
    object Loading : HomeScreenState
    data class Success(val decks: List<DeckEntity>) : HomeScreenState
    data class Error(val message: String) : HomeScreenState
}
