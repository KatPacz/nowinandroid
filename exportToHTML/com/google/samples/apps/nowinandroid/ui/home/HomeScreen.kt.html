// NOTICE: We have removed the "class HomeScreen {" wrapper
@Composable
fun HomeScreen(
    viewModel: HomeViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsState()

    // We'll use a Scaffold for basic layout structure
    Scaffold(
        topBar = { /* TopBar with search and filter will be added later */ },
        bottomBar = { /* Bottom navigation bar will be added later */ }
    ) { paddingValues ->
        when (val state = uiState) {
            is HomeScreenState.Loading -> {
                // Show a loading indicator
                Box(modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
                    CircularProgressIndicator()
                }
            }
            is HomeScreenState.Success -> {
                DeckGrid(decks = state.decks, modifier = Modifier.padding(paddingValues))
            }
            is HomeScreenState.Error -> {
                // Show an error message
                Text(text = "Error: ${state.message}")
            }
        }
    }
}

@Composable
fun DeckGrid(decks: List<DeckEntity>, modifier: Modifier = Modifier) {
    LazyVerticalGrid(
        columns = GridCells.Fixed(2), // 2 columns as per design
        modifier = modifier.padding(8.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp),
        horizontalArrangement = Arrangement.spacedBy(8.dp)

    ) {
        items(decks) { deck ->
            DeckTile(deck = deck)
        }
    }
}

@Composable
fun DeckTile(deck: DeckEntity) {
    // Basic representation of a deck tile
    Card(
        modifier = Modifier.aspectRatio(1f), // Makes it a square
        shape = RoundedCornerShape(16.dp),
        // The color will be dynamic based on category later
        colors = CardDefaults.cardColors(containerColor = Peach)
    ) {
        Box(modifier = Modifier.fillMaxSize().padding(16.dp), contentAlignment = Alignment.Center) {
            Text(
                text = deck.name,
                style = MaterialTheme.typography.titleLarge,
                textAlign = TextAlign.Center
            )
        }
    }
}
