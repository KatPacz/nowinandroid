package com.google.samples.apps.partygameapp.di

import android.content.Context
import com.google.samples.apps.partygameapp.data.database.AppDatabase
import dagger.Module
import dagger.Provides
import dagger.hilt.InstallIn
import dagger.hilt.android.qualifiers.ApplicationContext
import dagger.hilt.components.SingletonComponent
import javax.inject.Singleton

@Module
    @InstallIn(SingletonComponent::class)
    object AppModule {

        @Provides
        @Singleton
        fun provideAppDatabase(@ApplicationContext context: Context): AppDatabase {
            return Room.databaseBuilder(
                context,
                AppDatabase::class.java,
                "party_game_app_db"
            ).build()
        }

        @Provides
        @Singleton
        fun provideDeckDao(db: AppDatabase) = db.deckDao()

        // For now, let's create a placeholder repository
        @Provides
        @Singleton
        fun provideDeckRepository(deckDao: DeckDao): DeckRepository {
            return DeckRepository(deckDao)
        }
    }
