36   import com.yourcompany.partygameapp.ui.theme.PartyGameAppTheme
37   import dagger.hilt.android.AndroidEntryPoint

39   @AndroidEntryPoint
40   class MainActivity : ComponentActivity() {
41       override fun onCreate(savedInstanceState: Bundle?) {
42           super.onCreate(savedInstanceState)
43           installSplashScreen()
44           setContent {
45               PartyGameAppTheme {
46                   val navController = rememberNavController()
47                   val sharedViewModel: SharedGameState = hiltViewModel()
48                   val navBackStackEntry by navController.currentBackStackEntryAsState()
49                   val currentRoute = navBackStackEntry?.destination?.route
50   
51                   Scaffold(
52                       bottomBar = {
53                           if (currentRoute in listOf(Routes.HOME, Routes.IMPORT, Routes.LEADERBOARD, Routes.SETTINGS)) {
54                               BottomNavBar(onNavigate = { route ->
55                                   if (route != currentRoute) {
56                                       navController.navigate(route)
57                                   }
58                               })
59                           }
60                       }
61                   ) { padding ->
62                       NavHost(
63                           navController = navController,
64                           startDestination = Routes.HOME,
65                           modifier = Modifier.padding(padding)
66                       ) {
67                           composable(route = Routes.HOME) {
68                               HomeScreen(
69                                   onDeckSelected = { deckId -> navController.navigate("pre_game/$deckId") },
70                                   onNavigateToSettings = { navController.navigate(Routes.SETTINGS) },
71                                   onNavigateToLeaderboard = { navController.navigate(Routes.LEADERBOARD) }
72                               )
73                           }
74                           composable(
75                               route = Routes.PRE_GAME,
76                               arguments = listOf(navArgument(Routes.ARG_DECK_ID) { type = NavType.IntType })
77                           ) {
78                               val preGameViewModel: PreGameViewModel = hiltViewModel()
79                               PreGameScreen(
80                                   viewModel = preGameViewModel,
81                                   onStartGame = { gameMode, deckId, playerId ->
82                                       sharedViewModel.playerId = playerId
83                                       sharedViewModel.playerName = preGameViewModel.playerList.value.find { it.id == playerId }?.name ?: "Player"
84   
85                                       if (preGameViewModel.selectedGameMode == com.yourcompany.partygameapp.domain.model.GameMode.HotPotato) {
86                                           val teamTime = preGameViewModel.teamTimeSeconds
87                                           val maxSkips = preGameViewModel.maxSkipsPerTeam
88                                           val assignedPlayersJson = Gson().toJson(preGameViewModel.teamPlayers)
89                                           val encodedPlayersJson = Uri.encode(assignedPlayersJson)
90                                           navController.navigate("gameplay/$deckId/$playerId/$gameMode/$teamTime/$maxSkips/$encodedPlayersJson")
91                                       } else {
92                                           navController.navigate("gameplay/$deckId/$playerId/$gameMode")
93                                       }
94                                   },
95                                   // Use navigateUp for consistent back behavior
96                                   onNavigateBack = { navController.navigateUp() }
97                               )
98                           }
99                           composable(
100                              route = Routes.GAMEPLAY,
101                              arguments = listOf(
102                                  navArgument(Routes.ARG_DECK_ID) { type = NavType.IntType },
103                                  navArgument(Routes.ARG_PLAYER_ID) { type = NavType.IntType },
104                                  navArgument(Routes.ARG_GAME_MODE) { type = NavType.StringType },
105                                  navArgument(Routes.ARG_HOT_POTATO_TEAM_TIME) { type = NavType.IntType; defaultValue = 300 },
106                                  navArgument(Routes.ARG_HOT_POTATO_MAX_SKIPS) { type = NavType.IntType; defaultValue = 2 },
107                                  navArgument(Routes.ARG_HOT_POTATO_PLAYERS_JSON) { type = NavType.StringType; defaultValue = "[]" }
108                              )
109                          ) {
110                              GamePlayScreen(
111                                  onGameFinished = { results, redScore, blueScore, winningTeam ->
112                                      sharedViewModel.results = results
113                                      sharedViewModel.redTeamScore = redScore
114                                      sharedViewModel.blueTeamScore = blueScore
115                                      sharedViewModel.winningTeam = winningTeam
116                                      navController.navigate(Routes.AFTER_GAME) { popUpTo(Routes.PRE_GAME) { inclusive = true } }
117                                  }
118                              )
119                          }
120                          composable(route = Routes.AFTER_GAME) {
121                              val afterGameViewModel: AfterGameViewModel = hiltViewModel()
122                              LaunchedEffect(Unit) {
123                                  afterGameViewModel.saveGameResults(sharedViewModel.playerId, sharedViewModel.results)
124                              }
125                              AfterGameScreen(
126                                  results = sharedViewModel.results,
127                                  playerName = sharedViewModel.playerName,
128                                  redTeamScore = sharedViewModel.redTeamScore,
129                                  blueTeamScore = sharedViewModel.blueTeamScore,
130                                  winningTeam = sharedViewModel.winningTeam,
131                                  onPlayAgain = { navController.navigateUp() }, // Go back to PreGame to replay
132                                  onGoHome = { navController.navigate(Routes.HOME) { popUpTo(Routes.HOME) { inclusive = true } } }
133                              )
134                          }
135                          composable(Routes.LEADERBOARD) { LeaderboardScreen(onNavigateBack = { navController.navigateUp() }) }
136                          composable(Routes.SETTINGS) { SettingsScreen(onNavigateBack = { navController.navigateUp() }) }
137                          composable(Routes.IMPORT) { ImportScreen(onNavigateBack = { navController.navigateUp() }) }
138                          composable(Routes.EXPORT) { ExportScreen(onNavigateBack = { navController.navigateUp() }) }
139                      }
140                  }
141              }
142          }
143      }
144  }
