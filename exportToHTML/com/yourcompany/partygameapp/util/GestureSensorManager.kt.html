3    package com.yourcompany.partygameapp.util
4    
5    import android.content.Context
6    import android.hardware.Sensor
7    import android.hardware.SensorEvent
8    import android.hardware.SensorEventListener
9    import android.hardware.SensorManager
10   import kotlin.math.abs
11   
12   enum class GameGesture { TILT_FRONT, TILT_BACK, SHAKE }
13   
14   class GestureSensorManager(
15       context: Context,
16       private val onGestureDetected: (GameGesture) -> Unit
17   ) : SensorEventListener {
18   
19       private val sensorManager = context.getSystemService(Context.SENSOR_SERVICE) as SensorManager
20       private val accelerometer = sensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER)
21   
22       private val TILT_FORWARD_THRESHOLD = -5.0f
23       private val TILT_BACK_THRESHOLD = 5.0f
24       private val SHAKE_THRESHOLD = 1200f
25       private var lastUpdate: Long = 0
26       private var lastX = 0f; private var lastY = 0f; private var lastZ = 0f
27   
28       fun start() {
29           lastUpdate = System.currentTimeMillis()
30           sensorManager.registerListener(this, accelerometer, SensorManager.SENSOR_DELAY_GAME)
31       }
32   
33       fun stop() {
34           sensorManager.unregisterListener(this)
35       }
36   
37       override fun onAccuracyChanged(sensor: Sensor?, accuracy: Int) {}
38   
39       override fun onSensorChanged(event: SensorEvent) {
40           if (event.sensor.type != Sensor.TYPE_ACCELEROMETER) return
41           val currentTime = System.currentTimeMillis()
42           if ((currentTime - lastUpdate) > 100) { // Check every 100ms
43               val diffTime = (currentTime - lastUpdate)
44               lastUpdate = currentTime
45   
46               val x = event.values[0]; val y = event.values[1]; val z = event.values[2]
47   
48               if (z < TILT_FORWARD_THRESHOLD) {
49                   onGestureDetected(GameGesture.TILT_FRONT)
50                   return
51               }
52               if (z > TILT_BACK_THRESHOLD) {
53                   onGestureDetected(GameGesture.TILT_BACK)
54                   return
55               }
56   
57               val speed = abs(x + y + z - lastX - lastY - lastZ) / diffTime * 10000
58               if (speed > SHAKE_THRESHOLD) { onGestureDetected(GameGesture.SHAKE) }
59               lastX = x; lastY = y; lastZ = z
60           }
61       }
62   }
