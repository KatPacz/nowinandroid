3    package com.yourcompany.partygameapp.util
4    
5    import android.app.Application
6    import android.net.Uri
7    import android.provider.OpenableColumns
8    import java.io.FileOutputStream
9    import javax.inject.Inject
10   import javax.inject.Singleton
11   
12   @Singleton
13   class FileHelper @Inject constructor(
14       private val application: Application
15   ) {
16       /**
17        * Given a URI from the file picker, this function reads the actual file name.
18        */
19       fun getFileNameFromUri(uri: Uri): String? {
20           var fileName: String? = null
21           val cursor = application.contentResolver.query(uri, null, null, null, null)
22           cursor?.use {
23               if (it.moveToFirst()) {
24                   val nameIndex = it.getColumnIndex(OpenableColumns.DISPLAY_NAME)
25                   if (nameIndex != -1) {
26                       fileName = it.getString(nameIndex)
27                   }
28               }
29           }
30           return fileName
31       }
32   
33       /**
34        * Takes a list of strings and writes each one as a new line to the file
35        * represented by the given URI.
36        */
37       fun saveContentToUri(uri: Uri, content: List<String>) {
38           application.contentResolver.openFileDescriptor(uri, "w")?.use { parcelFileDescriptor ->
39               FileOutputStream(parcelFileDescriptor.fileDescriptor).use { fileOutputStream ->
40                   content.forEach { line ->
41                       fileOutputStream.write((line + "\n").toByteArray())
42                   }
43               }
44           }
45       }
46   }
