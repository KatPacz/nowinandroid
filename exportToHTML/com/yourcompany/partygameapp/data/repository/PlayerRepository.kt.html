3    package com.yourcompany.partygameapp.data.repository
4    
5    import com.yourcompany.partygameapp.data.database.dao.PlayerDao
6    import com.yourcompany.partygameapp.data.database.entity.PlayerEntity
7    import com.yourcompany.partygameapp.domain.model.Player
8    import kotlinx.coroutines.flow.Flow
9    import kotlinx.coroutines.flow.map
10   import javax.inject.Inject
11   import javax.inject.Singleton
12   
13   @Singleton
14   class PlayerRepository @Inject constructor(
15       private val playerDao: PlayerDao
16   ) {
17       fun getAllPlayers(): Flow<List<Player>> = playerDao.getAllPlayers().map { entities ->
18           entities.map { Player(id = it.id, name = it.name) }
19       }
20   
21       suspend fun insertPlayer(name: String) {
22           playerDao.insertPlayer(PlayerEntity(name = name))
23       }
24   
25       suspend fun getPlayerById(id: Int): PlayerEntity? {
26           return playerDao.getPlayerById(id)
27       }
28   
29       suspend fun updatePlayerStats(playerId: Int, correctGuesses: Int, score: Int) {
30           playerDao.updatePlayerStats(playerId, correctGuesses, score)
31       }
32   
33       fun getPlayersForLeaderboard(): Flow<List<PlayerEntity>> {
34           return playerDao.getAllPlayers()
35       }
36   }
