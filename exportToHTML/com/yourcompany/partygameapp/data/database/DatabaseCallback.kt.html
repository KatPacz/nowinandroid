3    package com.yourcompany.partygameapp.data.database
4    
5    import android.content.Context
6    import android.util.Log
7    import androidx.room.RoomDatabase
8    import androidx.sqlite.db.SupportSQLiteDatabase
9    import com.yourcompany.partygameapp.data.database.dao.CardDao
10   import com.yourcompany.partygameapp.data.database.dao.CategoryDao
11   import com.yourcompany.partygameapp.data.database.dao.DeckDao
12   import com.yourcompany.partygameapp.data.database.entity.CardEntity
13   import com.yourcompany.partygameapp.data.database.entity.CategoryEntity
14   import com.yourcompany.partygameapp.data.database.entity.DeckEntity
15   import kotlinx.coroutines.CoroutineScope
16   import kotlinx.coroutines.Dispatchers
17   import kotlinx.coroutines.launch
18   import java.io.IOException
19   import java.util.Locale
20   import javax.inject.Provider
21   
22   class DatabaseCallback(
23       private val context: Context,
24       private val categoryDaoProvider: Provider<CategoryDao>,
25       private val deckDaoProvider: Provider<DeckDao>,
26       private val cardDaoProvider: Provider<CardDao>
27   ) : RoomDatabase.Callback() {
28   
29       private val TAG = "DATABASE_DEBUG"
30   
31       override fun onOpen(db: SupportSQLiteDatabase) {
32           super.onOpen(db)
33           Log.d(TAG, "----------------------------------------------------")
34           Log.d(TAG, "DATABASE OPENED. Checking if population is needed.")
35           Log.d(TAG, "----------------------------------------------------")
36           CoroutineScope(Dispatchers.IO).launch {
37               checkAndPopulate()
38           }
39       }
40   
41       private suspend fun checkAndPopulate() {
42           try {
43               if (deckDaoProvider.get().getDeckCount() == 0) {
44                   Log.d(TAG, "DATABASE IS EMPTY. Starting population process...")
45                   populateFromAssets()
46               } else {
47                   Log.d(TAG, "Database already has data. Skipping population.")
48               }
49           } catch (e: Exception) {
50               Log.e(TAG, "CRITICAL ERROR in checkAndPopulate", e)
51           }
52       }
53   
54       private suspend fun populateFromAssets() {
55           Log.d(TAG, "-> Starting populateFromAssets().")
56           val deckDao = deckDaoProvider.get()
57           val categoryDao = categoryDaoProvider.get()
58           val cardDao = cardDaoProvider.get()
59           val assetManager = context.assets
60   
61           val colorPalette = listOf("#F4A261", "#6B8E85", "#4A5859", "#2A2D43", "#E9C46A", "#F4A261", "#E76F51")
62           var colorIndex = 0
63   
64           try {
65               val langFolders = assetManager.list("decks")
66               if (langFolders.isNullOrEmpty()) {
67                   Log.e(TAG, "CRITICAL ERROR: 'decks' folder in assets is empty or does not exist.")
68                   return
69               }
70               Log.d(TAG, "Found language folders: ${langFolders.joinToString()}")
71   
72               for (langFolder in langFolders) {
73                   val deckFiles = assetManager.list("decks/$langFolder")
74                   if (deckFiles.isNullOrEmpty()) {
75                       Log.w(TAG, "Warning: Language folder 'decks/$langFolder' is empty.")
76                       continue
77                   }
78                   Log.d(TAG, "In '$langFolder', found deck files: ${deckFiles.joinToString()}")
79   
80                   for (fileName in deckFiles) {
81                       if (!fileName.endsWith(".txt")) continue
82                       val categoryName = fileName.removeSuffix(".txt").replaceFirstChar { it.titlecase(Locale.ROOT) }
83                       val color = colorPalette[colorIndex % colorPalette.size]
84                       colorIndex++
85   
86                       Log.d(TAG, "Processing: Lang='$langFolder', Category='$categoryName'")
87                       val categoryId = categoryDao.insertCategory(CategoryEntity(name = categoryName, colorHex = color))
88                       Log.d(TAG, "  - Inserted category '$categoryName' with ID: $categoryId")
89   
90                       val languageCode = langFolder.lowercase(Locale.ROOT)
91                       val deckId = deckDao.insertDeck(DeckEntity(
92                           name = categoryName,
93                           description = "A collection of $categoryName",
94                           categoryId = categoryId.toInt(),
95                           languageCode = languageCode
96                       ))
97                       Log.d(TAG, "  - Inserted deck '$categoryName' with ID: $deckId for lang '$languageCode'")
98   
99                       val path = "decks/$langFolder/$fileName"
100                      assetManager.open(path).bufferedReader().useLines { lines ->
101                          lines.filter { it.isNotBlank() }.forEach { line ->
102                              cardDao.insertCard(CardEntity(text = line, deckId = deckId.toInt()))
103                          }
104                      }
105                      Log.d(TAG, "  - Successfully inserted cards from $path")
106                  }
107              }
108              Log.d(TAG, "-> DATABASE POPULATION COMPLETE.")
109          } catch (e: Exception) {
110              Log.e(TAG, "CRITICAL UNEXPECTED ERROR during population", e)
111          }
112      }
113  }
