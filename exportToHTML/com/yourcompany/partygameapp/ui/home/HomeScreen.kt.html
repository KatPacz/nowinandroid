39       onNavigateToLeaderboard: () -> Unit, // <<< ADD THIS PARAMETER
40       viewModel: HomeViewModel = HiltViewModel() as HomeViewModel
41   ) {
42       val uiState by viewModel.uiState.collectAsState()

45       Scaffold(
46           topBar = {
47               TopAppBar(
48                   title = { Text(stringResource(R.string.home_title)) },
49                   actions = {
50                       // Leaderboard Button
51                       IconButton(onClick = onNavigateToLeaderboard) {
52                           Icon(Icons.Filled.Leaderboard, contentDescription = stringResource(R.string.leaderboard_title))

54                       // Settings Button
55                       IconButton(onClick = onNavigateToSettings) {
56                           Icon(Icons.Default.Settings, contentDescription = stringResource(R.string.settings_title))
57                       }
58                       // Buy a Coffee button
59                       IconButton(
60                           onClick = {
61                               val intent = Intent(Intent.ACTION_VIEW, Uri.parse("https://ko-fi.com/your_profile"))
62                               context.startActivity(intent)
63                           }
64                       ) {
65                           Icon(Icons.Default.Coffee, contentDescription = stringResource(R.string.home_buy_coffee_button))
66                       }
67   
68                       LanguageDropDownMenu(viewModel = viewModel)
69                   }
70               )
71           }
72       ) { padding ->
73           Box(modifier = Modifier.padding(padding).fillMaxSize()) {
74               when (val state = uiState) {
75                   is HomeScreenUiState.Loading -> CircularProgressIndicator(modifier = Modifier.align(Alignment.Center))
76                   is HomeScreenUiState.Success -> {
77                       if (state.decks.isEmpty()) {
78                           Text(
79                               text = stringResource(R.string.home_no_decks),
80                               modifier = Modifier.align(Alignment.Center),
81                               style = MaterialTheme.typography.bodyLarge
82                           )
83                       } else {
84                           DeckGrid(decks = state.decks, onDeckSelected = onDeckSelected)
85                       }
86                   }
87               }
88           }
89       }

91   @Composable
92   fun LanguageDropDownMenu(viewModel: HomeViewModel) {
93       val selectedLanguage by viewModel.selectedLanguage.collectAsState()
94       val isMenuExpanded by viewModel.isMenuExpanded.collectAsState()
95   
96       Box {
97           TextButton(onClick = { viewModel.onLanguageMenuClick() }) {
98               Text(
99                   text = selectedLanguage.uppercase(Locale.ROOT),
100                  style = MaterialTheme.typography.titleMedium
101              )
102          }
103          DropdownMenu(
104              expanded = isMenuExpanded,
105              onDismissRequest = { viewModel.onLanguageMenuDismiss() }
106          ) {
107              viewModel.availableLanguages.forEach { languageCode ->
108                  DropdownMenuItem(
109                      text = { Text(languageCode.uppercase(Locale.ROOT)) },
110                      onClick = { viewModel.onLanguageSelected(languageCode) }
111                  )
112              }
113          }
114      }
115  }

117  @Composable
118  fun DeckGrid(
119      decks: List<DeckWithCategory>,
120      onDeckSelected: (Int) -> Unit,
121      modifier: Modifier = Modifier
122  ) {
123      LazyVerticalGrid(
124          columns = GridCells.Fixed(2),
125          modifier = modifier.padding(8.dp),
126          verticalArrangement = Arrangement.spacedBy(8.dp),
127          horizontalArrangement = Arrangement.spacedBy(8.dp)
128      ) {
129          items(decks) { deck ->
130              DeckTile(deck = deck, onClick = { onDeckSelected(deck.deckId) })
131          }
132      }
133  }

135  @Composable
136  fun DeckTile(deck: DeckWithCategory, onClick: () -> Unit) {
137      Card(
138          modifier = Modifier
139              .aspectRatio(1f)
140              .clickable(onClick = onClick),
141          shape = RoundedCornerShape(16.dp),
142          colors = CardDefaults.cardColors(containerColor = Color(android.graphics.Color.parseColor(deck.categoryColorHex)))
143      ) {
144          Box(
145              modifier = Modifier
146                  .fillMaxSize()
147                  .padding(16.dp),
148              contentAlignment = Alignment.Center
149          ) {
150              Text(
151                  text = deck.deckName,
152                  style = MaterialTheme.typography.titleLarge,
153                  textAlign = TextAlign.Center
154              )
155          }
156      }
