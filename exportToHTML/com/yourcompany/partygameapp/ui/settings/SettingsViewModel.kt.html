3    package com.yourcompany.partygameapp.ui.settings
4    
5    import androidx.lifecycle.ViewModel
6    import androidx.lifecycle.viewModelScope
7    import com.yourcompany.partygameapp.data.repository.SettingsRepository
8    import com.yourcompany.partygameapp.domain.model.GameSettings
9    import dagger.hilt.android.lifecycle.HiltViewModel
10   import kotlinx.coroutines.flow.SharingStarted
11   import kotlinx.coroutines.flow.StateFlow
12   import kotlinx.coroutines.flow.stateIn
13   import kotlinx.coroutines.launch
14   import javax.inject.Inject
15   
16   @HiltViewModel
17   class SettingsViewModel @Inject constructor(
18       private val settingsRepository: SettingsRepository
19   ) : ViewModel() {
20   
21       val settingsState: StateFlow<GameSettings?> = settingsRepository.gameSettingsFlow
22           .stateIn(
23               scope = viewModelScope,
24               started = SharingStarted.WhileSubscribed(5000),
25               initialValue = null
26           )
27   
28       fun onTimeUpPenaltyChanged(isEnabled: Boolean) {
29           viewModelScope.launch {
30               settingsRepository.setTimeUpPenalty(isEnabled)
31           }
32       }
33   
34       fun onGiveUpPenaltyChanged(isEnabled: Boolean) {
35           viewModelScope.launch {
36               settingsRepository.setGiveUpPenalty(isEnabled)
37           }
38       }
39   
40       fun onCardTimeChanged(seconds: Int) {
41           viewModelScope.launch {
42               val clampedValue = seconds.coerceIn(10, 120)
43               settingsRepository.setCardTime(clampedValue)
44           }
45       }
46   
47       fun onCardCountChanged(count: Int) {
48           viewModelScope.launch {
49               val clampedValue = count.coerceIn(5, 20)
50               settingsRepository.setCardCount(clampedValue)
51           }
52       }
53   }
