3    package com.yourcompany.partygameapp.ui.leaderboard
4    
5    import androidx.lifecycle.ViewModel
6    import androidx.lifecycle.viewModelScope
7    import com.yourcompany.partygameapp.data.database.entity.PlayerEntity
8    import com.yourcompany.partygameapp.data.repository.PlayerRepository
9    import dagger.hilt.android.lifecycle.HiltViewModel
10   import kotlinx.coroutines.flow.SharingStarted
11   import kotlinx.coroutines.flow.StateFlow
12   import kotlinx.coroutines.flow.map
13   import kotlinx.coroutines.flow.stateIn
14   import javax.inject.Inject
15   
16   @HiltViewModel
17   class LeaderboardViewModel @Inject constructor(
18       playerRepository: PlayerRepository
19   ) : ViewModel() {
20       val players: StateFlow<List<PlayerEntity>> = playerRepository.getPlayersForLeaderboard()
21           .map { list -> list.sortedByDescending { it.totalPoints } }
22           .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
23   }
