18   import com.yourcompany.partygameapp.ui.navigation.Routes

20   sealed class BottomNavItem(val route: String, val icon: ImageVector, val stringResId: Int) { // Use stringResId
21       object Home : BottomNavItem(Routes.HOME, Icons.Default.Home, R.string.home_title)
22       object Import : BottomNavItem(Routes.IMPORT, Icons.Default.AddCircle, R.string.import_title)
23       object Leaderboard : BottomNavItem(Routes.LEADERBOARD, Icons.Default.Leaderboard, R.string.leaderboard_title)
24       object Settings : BottomNavItem(Routes.SETTINGS, Icons.Default.Settings, R.string.settings_title)
25   }
26   
27   @Composable
28   fun BottomNavBar(
29       onNavigate: (String) -> Unit
30   ) {
31       val items = listOf(
32           BottomNavItem.Home,
33           BottomNavItem.Import,
34           BottomNavItem.Leaderboard,
35           BottomNavItem.Settings
36       )
37   
38       NavigationBar {
39           items.forEach { item ->
40               NavigationBarItem(
41                   icon = { Icon(item.icon, contentDescription = stringResource(item.stringResId)) },
42                   label = { Text(stringResource(item.stringResId)) },
43                   selected = false,
44                   onClick = { onNavigate(item.route) }
45               )
46           }
47       }
48   }
