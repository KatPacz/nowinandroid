34   @OptIn(ExperimentalMaterial3Api::class)
35   @Composable
36   fun ExportScreen(
37       onNavigateBack: () -> Unit,
38       viewModel: ImportExportViewModel = hiltViewModel()
39   ) {
40       val decks by viewModel.allDecksForExport.collectAsState()
41       val deckToExport by viewModel.deckToExport.collectAsState()
42       val context = LocalContext.current
43   
44       val fileSaverLauncher = rememberLauncherForActivityResult(
45           contract = ActivityResultContracts.CreateDocument("text/plain"),
46           onResult = { uri ->
47               uri?.let { safeUri ->
48                   deckToExport?.let { data ->
49                       val hiltEntryPoint = EntryPointAccessors.fromActivity(
50                           context as android.app.Activity,
51                           FileHelperEntryPoint::class.java
52                       )
53                       val fileHelper = hiltEntryPoint.getFileHelper()
54                       fileHelper.saveContentToUri(safeUri, data.content)
55                   }
56               }
57               viewModel.onFileExported()
58           }
59       )
60   
61       LaunchedEffect(deckToExport) {
62           deckToExport?.let {
63               fileSaverLauncher.launch(it.suggestedName)
64           }
65       }
66   
67       Scaffold(
68           topBar = {
69               TopAppBar(
70                   title = { Text(stringResource(R.string.export_title)) },
71                   navigationIcon = {
72                       IconButton(onClick = onNavigateBack) {
73                           Icon(Icons.AutoMirrored.Filled.ArrowBack, contentDescription = stringResource(android.R.string.cancel))
74                       }
75                   }
76               )
77           }
78       ) { padding ->
79           Column(modifier = Modifier.padding(padding).fillMaxSize()) {
80               if (decks.isEmpty()) {
81                   // <<< START OF NEW CODE: Empty State >>>
82                   Box(modifier = Modifier.fillMaxSize().weight(1f), contentAlignment = Alignment.Center) {
83                       Text(
84                           text = stringResource(R.string.export_no_decks_found),
85                           style = MaterialTheme.typography.titleMedium,
86                           textAlign = TextAlign.Center
87                       )
88                   }
89                   // <<< END OF NEW CODE >>>
90               } else {
91                   Row(
92                       modifier = Modifier.fillMaxWidth().padding(horizontal = 16.dp),
93                       horizontalArrangement = Arrangement.SpaceBetween,
94                       verticalAlignment = Alignment.CenterVertically
95                   ) {
96                       Text(stringResource(R.string.export_select_decks), style = MaterialTheme.typography.titleMedium)
97                       Row {
98                           TextButton(onClick = { viewModel.onSelectAll(decks) }) { Text(stringResource(R.string.export_all_button)) }
99                           TextButton(onClick = viewModel::onDeselectAll) { Text(stringResource(R.string.export_none_button)) }
100                      }
101                  }
102                  HorizontalDivider()
103                  LazyColumn(modifier = Modifier.weight(1f)) {
104                      items(decks) { deck ->
105                          DeckExportRow(
106                              deck = deck,
107                              isChecked = viewModel.checkedDeckIds[deck.deckId] ?: false,
108                              onCheckedChanged = { isChecked ->
109                                  viewModel.onDeckCheckedChanged(deck.deckId, isChecked)
110                              }
111                          )
112                      }
113                  }
114              }
115              Button(
116                  onClick = viewModel::onExport,
117                  enabled = viewModel.checkedDeckIds.any { it.value },
118                  modifier = Modifier.fillMaxWidth().padding(16.dp).height(56.dp)
119              ) {
120                  Text(stringResource(R.string.export_button), style = MaterialTheme.typography.titleLarge)
121              }
122          }
123      }
124  }
