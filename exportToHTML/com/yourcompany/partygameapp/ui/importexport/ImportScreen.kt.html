108  @OptIn(ExperimentalMaterial3Api::class)
109  @Composable
110  fun CategoryDropdown(
111      categories: List<String>,
112      selectedCategory: String,
113      onCategoryChange: (String) -> Unit
114  ) {
115      var isExpanded by remember { mutableStateOf(false) }
116  
117      ExposedDropdownMenuBox(
118          expanded = isExpanded,
119          onExpandedChange = { isExpanded = !isExpanded }
120      ) {
121          TextField(
122              value = selectedCategory,
123              onValueChange = onCategoryChange,
124              label = { Text(stringResource(R.string.import_category_label)) },
125              trailingIcon = { ExposedDropdownMenuDefaults.TrailingIcon(expanded = isExpanded) },
126              modifier = Modifier
127                  .menuAnchor()
128                  .fillMaxWidth()
129          )
130  
131          val filteredOptions = categories.filter {
132              it.contains(selectedCategory, ignoreCase = true)
133          }
134  
135          if (filteredOptions.isNotEmpty()) {
136              ExposedDropdownMenu(
137                  expanded = isExpanded,
138                  onDismissRequest = { isExpanded = false }
139              ) {
140                  filteredOptions.forEach { categoryName ->
141                      DropdownMenuItem(
142                          text = { Text(categoryName) },
143                          onClick = {
144                              onCategoryChange(categoryName)
145                              isExpanded = false
146                          }
147                      )
148                  }
149              }
150          }
151      }
