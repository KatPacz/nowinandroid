3    package com.yourcompany.partygameapp.ui.importexport
4    
5    import android.content.Context
6    import android.net.Uri
7    import android.widget.Toast
8    import androidx.compose.runtime.getValue
9    import androidx.compose.runtime.mutableStateMapOf
10   import androidx.compose.runtime.mutableStateOf
11   import androidx.compose.runtime.setValue
12   import androidx.lifecycle.ViewModel
13   import androidx.lifecycle.viewModelScope
14   import com.yourcompany.partygameapp.data.database.entity.CategoryEntity
15   import com.yourcompany.partygameapp.data.repository.DeckRepository
16   import com.yourcompany.partygameapp.domain.model.DeckWithCategory
17   import com.yourcompany.partygameapp.util.FileHelper
18   import dagger.hilt.android.lifecycle.HiltViewModel
19   import dagger.hilt.android.qualifiers.ApplicationContext
20   import kotlinx.coroutines.flow.*
21   import kotlinx.coroutines.launch
22   import java.io.BufferedReader
23   import java.io.InputStreamReader
24   import javax.inject.Inject
25   
26   data class DeckToExport(val suggestedName: String, val content: List<String>)
27   
28   @HiltViewModel
29   class ImportExportViewModel @Inject constructor(
30       private val deckRepository: DeckRepository,
31       private val fileHelper: FileHelper,
32       @ApplicationContext private val context: Context
33   ) : ViewModel() {
34   
35       // --- IMPORT STATE ---
36       var selectedFileUri by mutableStateOf<Uri?>(null)
37           private set
38       var selectedFileName by mutableStateOf<String?>(null)
39           private set
40       var newDeckName by mutableStateOf("")
41           private set
42       var selectedCategoryName by mutableStateOf("")
43           private set
44   
45       val categories: StateFlow<List<CategoryEntity>> = deckRepository.getAllCategories()
46           .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
47   
48       fun onFileSelected(uri: Uri?) {
49           selectedFileUri = uri
50           if (uri != null) {
51               val fileName = fileHelper.getFileNameFromUri(uri)
52               selectedFileName = fileName
53               newDeckName = fileName?.removeSuffix(".txt")?.removeSuffix(".csv") ?: ""
54           } else {
55               selectedFileName = null
56               newDeckName = ""
57           }
58       }
59       fun onDeckNameChange(name: String) { newDeckName = name }
60       fun onCategoryChange(name: String) { selectedCategoryName = name }
61   
62       fun onSave(languageCode: String) {
63           val uri = selectedFileUri ?: return
64           if (newDeckName.isBlank() || selectedCategoryName.isBlank()) {
65               Toast.makeText(context, "Deck name and category cannot be empty", Toast.LENGTH_SHORT).show()
66               return
67           }
68           viewModelScope.launch {
69               try {
70                   val lines = readFileContent(uri)
71                   deckRepository.importNewDeck(newDeckName.trim(), selectedCategoryName.trim(), languageCode, lines)
72                   Toast.makeText(context, "'$newDeckName' imported successfully!", Toast.LENGTH_SHORT).show()
73                   resetState()
74               } catch (e: Exception) {
75                   Toast.makeText(context, "Error: ${e.message}", Toast.LENGTH_LONG).show()
76               }
77           }
78       }
79   
80       private fun readFileContent(uri: Uri): List<String> {
81           val stringList = mutableListOf<String>()
82           context.contentResolver.openInputStream(uri)?.use { inputStream ->
83               BufferedReader(InputStreamReader(inputStream)).use { reader ->
84                   reader.readLines().forEach { stringList.add(it) }
85               }
86           }
87           return stringList
88       }
89   
90       private fun resetState() {
91           selectedFileUri = null
92           selectedFileName = null
93           newDeckName = ""
94           selectedCategoryName = ""
95       }
96   
97   
98       // --- EXPORT STATE ---
99       val allDecksForExport: StateFlow<List<DeckWithCategory>> = deckRepository.getAllDecksForExport()
100          .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
101      val checkedDeckIds = mutableStateMapOf<Int, Boolean>()
102  
103      private val _deckToExport = MutableStateFlow<DeckToExport?>(null)
104      val deckToExport: StateFlow<DeckToExport?> = _deckToExport
105  
106  
107      // --- EXPORT EVENTS ---
108      fun onDeckCheckedChanged(deckId: Int, isChecked: Boolean) { checkedDeckIds[deckId] = isChecked }
109      fun onSelectAll(decks: List<DeckWithCategory>) { decks.forEach { checkedDeckIds[it.deckId] = true } }
110      fun onDeselectAll() { checkedDeckIds.clear() }
111  
112      fun onExport() {
113          val selectedIds = checkedDeckIds.filter { it.value }.keys
114          if (selectedIds.isEmpty()) {
115              Toast.makeText(context, "No decks selected", Toast.LENGTH_SHORT).show()
116              return
117          }
118          viewModelScope.launch {
119              val firstDeckId = selectedIds.first()
120              val deckInfo = allDecksForExport.value.find { it.deckId == firstDeckId } ?: return@launch
121  
122              val cards = deckRepository.getCardsForDeck(firstDeckId)
123              val cardLines = cards.map { it.text }
124  
125              _deckToExport.value = DeckToExport(
126                  suggestedName = "${deckInfo.deckName}.txt",
127                  content = cardLines
128              )
129          }
130      }
131  
132      fun onFileExported() {
133          _deckToExport.value = null
134          checkedDeckIds.clear()
135          Toast.makeText(context, "Deck exported successfully!", Toast.LENGTH_SHORT).show()
136      }
137  }
