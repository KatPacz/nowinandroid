3    package com.yourcompany.partygameapp.ui.aftergame
4    
5    import androidx.lifecycle.ViewModel
6    import androidx.lifecycle.viewModelScope
7    import com.yourcompany.partygameapp.data.repository.PlayerRepository
8    import com.yourcompany.partygameapp.domain.model.CardResult
9    import dagger.hilt.android.lifecycle.HiltViewModel
10   import kotlinx.coroutines.launch
11   import javax.inject.Inject
12   
13   @HiltViewModel
14   class AfterGameViewModel @Inject constructor(
15       private val playerRepository: PlayerRepository
16   ) : ViewModel() {
17   
18       fun saveGameResults(playerId: Int, results: List<CardResult>) {
19           if (playerId == 0) return // Don't save stats for guest player
20   
21           viewModelScope.launch {
22               val score = results.sumOf { it.points }.coerceAtLeast(0)
23               val correctGuesses = results.count { it.points > 0 }
24               playerRepository.updatePlayerStats(playerId, correctGuesses, score)
25           }
26       }
27   }
