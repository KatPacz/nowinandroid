24   @OptIn(ExperimentalMaterial3Api::class)
25   @Composable
26   fun AfterGameScreen(
27       results: List<CardResult>,
28       playerName: String,
29       redTeamScore: Int? = null, // Hot Potato specific
30       blueTeamScore: Int? = null, // Hot Potato specific
31       winningTeam: Team? = null, // Hot Potato specific
32       onPlayAgain: () -> Unit,
33       onGoHome: () -> Unit
34   ) {
35       val totalScore = results.sumOf { it.points }.coerceAtLeast(0)
36       val correctAnswers = results.count { it.points > 0 }
37       val skippedOrFailed = results.count { it.points <= 0 }
38   
39       Scaffold(
40           topBar = { TopAppBar(title = { Text(stringResource(R.string.aftergame_title)) }) }
41       ) { padding ->
42           Column(
43               modifier = Modifier.padding(padding).fillMaxSize().padding(16.dp),
44               horizontalAlignment = Alignment.CenterHorizontally
45           ) {
46               // Display winner for Hot Potato, otherwise general congrats
47               if (winningTeam != null && winningTeam != Team.NONE) {
48                   Icon(
49                       imageVector = Icons.Filled.EmojiEvents,
50                       contentDescription = stringResource(R.string.aftergame_winning_team_label, winningTeam.name),
51                       modifier = Modifier.size(64.dp),
52                       tint = if (winningTeam == Team.RED) Color.Red else Color.Blue
53                   )
54                   Spacer(Modifier.height(8.dp))
55                   Text(
56                       text = stringResource(R.string.aftergame_winning_team_label, winningTeam.name),
57                       style = MaterialTheme.typography.headlineMedium,
58                       fontWeight = FontWeight.Bold,
59                       color = if (winningTeam == Team.RED) Color.Red else Color.Blue
60                   )
61               } else {
62                   Text(stringResource(R.string.aftergame_congrats, playerName), style = MaterialTheme.typography.headlineMedium)
63               }
64   
65               Spacer(Modifier.height(16.dp))
66   
67               // Display scores based on game mode
68               if (redTeamScore != null && blueTeamScore != null) {
69                   // Hot Potato Score Display
70                   Row(verticalAlignment = Alignment.CenterVertically, horizontalArrangement = Arrangement.SpaceAround, modifier = Modifier.fillMaxWidth()) {
71                       Text(stringResource(R.string.hot_potato_red_team) + " $redTeamScore", style = MaterialTheme.typography.titleLarge, color = Color.Red, fontWeight = FontWeight.Bold)
72                       Text(stringResource(R.string.hot_potato_blue_team) + " $blueTeamScore", style = MaterialTheme.typography.titleLarge, color = Color.Blue, fontWeight = FontWeight.Bold)
73                   }
74               } else {
75                   // Standard/3Lives Score Display
76                   Row(verticalAlignment = Alignment.CenterVertically) {
77                       Icon(imageVector = Icons.Filled.CheckCircle, contentDescription = stringResource(R.string.aftergame_correct_label), tint = Color.Green)
78                       Text(" $correctAnswers", style = MaterialTheme.typography.titleLarge)
79                       Spacer(Modifier.width(24.dp))
80                       Icon(imageVector = Icons.Filled.Dangerous, contentDescription = stringResource(R.string.aftergame_incorrect_label), tint = Color.Red)
81                       Text(" $skippedOrFailed", style = MaterialTheme.typography.titleLarge)
82                   }
83                   Spacer(Modifier.height(8.dp))
84                   Text(stringResource(R.string.aftergame_score) + " $totalScore", style = MaterialTheme.typography.displaySmall, fontWeight = FontWeight.Bold)
85               }
86   
87               HorizontalDivider(modifier = Modifier.padding(vertical = 16.dp))
88   
89               LazyColumn(modifier = Modifier.weight(1f)) {
90                   items(results) { result -> ResultRow(result = result) }
91               }
92               Spacer(Modifier.height(16.dp))
93               Row(modifier = Modifier.fillMaxWidth()) {
94                   Button(onClick = onGoHome, modifier = Modifier.weight(1f)) { Text(stringResource(R.string.aftergame_main_menu_button)) }
95                   Spacer(Modifier.width(8.dp))
96                   Button(onClick = onPlayAgain, modifier = Modifier.weight(1f)) { Text(stringResource(R.string.aftergame_play_again_button)) }
97               }
98           }
99       }
100  }
101  
102  @Composable
103  fun ResultRow(result: CardResult) {
104      Row(
105          modifier = Modifier.fillMaxWidth().padding(vertical = 8.dp),
106          verticalAlignment = Alignment.CenterVertically,
107          horizontalArrangement = Arrangement.SpaceBetween
108      ) {
109          Text(result.cardText, style = MaterialTheme.typography.bodyLarge)
110          Text(
111              text = when {
112                  result.points > 0 -> "+${result.points}"
113                  else -> "${result.points}"
114              },
115              color = if (result.points > 0) Color.Green else if (result.points < 0) Color.Red else Color.Gray,
116              fontWeight = FontWeight.Bold
117          )
118      }
119  }
