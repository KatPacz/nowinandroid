21   @OptIn(ExperimentalMaterial3Api::class)
22   @Composable
23   fun PreGameScreen(
24       onStartGame: (String, Int, Int) -> Unit, // gameMode (String), deckId (Int), playerId (Int)
25       onNavigateBack: () -> Unit,
26       viewModel: PreGameViewModel = hiltViewModel()
27   ) {
28       val deck by viewModel.selectedDeck.collectAsState()
29       val players by viewModel.playerList.collectAsState()
30   
31       Scaffold(
32           topBar = { TopAppBar(title = { deck?.name?.let { Text(it) } ?: Text("Loading...") }) }
33       ) { padding ->
34           Column(
35               modifier = Modifier.padding(padding).fillMaxSize().padding(16.dp),
36               horizontalAlignment = Alignment.CenterHorizontally
37           ) {
38               if (deck == null) {
39                   CircularProgressIndicator()
40               } else {
41                   DeckInfoCard(deck = deck!!)
42               }
43   
44               Spacer(modifier = Modifier.height(24.dp))
45               Text("Select game mode", style = MaterialTheme.typography.titleMedium)
46               Spacer(modifier = Modifier.height(8.dp))
47               GameModeDropdown(
48                   modes = viewModel.availableGameModes,
49                   selectedMode = viewModel.selectedGameMode,
50                   onModeSelected = viewModel::onGameModeSelected,
51                   isExpanded = viewModel.isGameModeDropdownExpanded,
52                   onDropdownClick = viewModel::onGameModeDropdownClick,
53                   onDropdownDismiss = viewModel::onGameModeDropdownDismiss
54               )
55   
56               // --- HOT POTATO SETTINGS (Only visible when HotPotato is selected) ---
57               if (viewModel.selectedGameMode == GameMode.HotPotato) {
58                   Spacer(modifier = Modifier.height(24.dp))
59                   Text("Hot Potato Settings", style = MaterialTheme.typography.titleMedium)
60                   Spacer(modifier = Modifier.height(8.dp))
61                   SettingStepper(
62                       title = stringResource(R.string.hot_potato_time_per_team),
63                       value = viewModel.teamTimeSeconds,
64                       onValueChange = viewModel::onTeamTimeChange,
65                       range = 60..600, // 1-10 minutes
66                       step = 30, // 30 seconds increment
67                       unit = stringResource(R.string.settings_unit_seconds)
68                   )
69                   SettingStepper(
70                       title = stringResource(R.string.hot_potato_max_skips_per_team),
71                       value = viewModel.maxSkipsPerTeam,
72                       onValueChange = viewModel::onMaxSkipsChange,
73                       range = 1..5,
74                       step = 1,
75                       unit = ""
76                   )
77                   Spacer(modifier = Modifier.height(16.dp))
78                   Button(onClick = viewModel::onAssignPlayersToTeams, modifier = Modifier.fillMaxWidth()) {
79                       Text(stringResource(R.string.hot_potato_assign_players_button))
80                   }
81                   Spacer(modifier = Modifier.height(8.dp))
82                   TeamAssignments(players = players, teamPlayers = viewModel.teamPlayers)
83               }
84               // --- END HOT POTATO SETTINGS ---
85   
86               Spacer(modifier = Modifier.height(24.dp))
87               Text(stringResource(R.string.pregame_pick_or_add_name), style = MaterialTheme.typography.titleMedium)
88               Spacer(modifier = Modifier.height(8.dp))
89               PlayerDropdown(
90                   players = players,
91                   selectedPlayer = viewModel.selectedPlayer,
92                   onPlayerSelected = viewModel::onPlayerSelected,
93                   isExpanded = viewModel.isPlayerDropdownExpanded,
94                   onDropdownClick = viewModel::onDropdownClick,
95                   onDropdownDismiss = viewModel::onDropdownDismiss
96               )
97   
98               Spacer(modifier = Modifier.height(16.dp))
99   
100              OutlinedTextField(
101                  value = viewModel.newPlayerName,
102                  onValueChange = viewModel::onNewPlayerNameChange,
103                  label = { Text(stringResource(R.string.pregame_add_new_player)) },
104                  modifier = Modifier.fillMaxWidth()
105              )
106              Spacer(modifier = Modifier.height(8.dp))
107              Button(onClick = viewModel::onAddPlayer, modifier = Modifier.fillMaxWidth()) {
108                  Text(stringResource(R.string.pregame_add_player_button))
109              }
110  
111              Spacer(modifier = Modifier.weight(1f))
112              Button(
113                  onClick = {
114                      deck?.id?.let { deckId ->
115                          viewModel.selectedPlayer?.id?.let { playerId ->
116                              onStartGame(viewModel.selectedGameMode.name, deckId, playerId)
117                          }
118                      }
119                  },
120                  enabled = viewModel.selectedPlayer != null && (viewModel.selectedGameMode != GameMode.HotPotato || viewModel.teamPlayers.isNotEmpty()),
121                  modifier = Modifier.fillMaxWidth().height(56.dp)
122              ) {
123                  Text(stringResource(R.string.pregame_start_button), style = MaterialTheme.typography.titleLarge)
124              }
125          }
126      }
127  }

141  @OptIn(ExperimentalMaterial3Api::class)
142  @Composable
143  fun PlayerDropdown(
144      players: List<Player>,
145      selectedPlayer: Player?,
146      onPlayerSelected: (Player) -> Unit,
147      isExpanded: Boolean,
148      onDropdownClick: () -> Unit,
149      onDropdownDismiss: () -> Unit
150  ) {
151      ExposedDropdownMenuBox(
152          expanded = isExpanded,
153          onExpandedChange = { onDropdownClick() }
154      ) {
155          OutlinedTextField(
156              value = selectedPlayer?.name ?: stringResource(R.string.pregame_select_player_placeholder), // Use stringResource
157              onValueChange = {},
158              readOnly = true,
159              trailingIcon = { ExposedDropdownMenuDefaults.TrailingIcon(expanded = isExpanded) },
160              modifier = Modifier.menuAnchor().fillMaxWidth()
161          )
162          ExposedDropdownMenu(
163              expanded = isExpanded,
164              onDismissRequest = onDropdownDismiss
165          ) {
166              players.forEach { player ->
167                  DropdownMenuItem(
168                      text = { Text(player.name) },
169                      onClick = { onPlayerSelected(player) }
170                  )
171              }
172          }
173      }
174  }

211  @Composable
212  fun TeamAssignments(players: List<Player>, teamPlayers: Map<Int, Team>) {
213      val redTeam = players.filter { teamPlayers[it.id] == Team.RED }
214      val blueTeam = players.filter { teamPlayers[it.id] == Team.BLUE }
215  
216      if (redTeam.isNotEmpty() || blueTeam.isNotEmpty()) {
217          Column(modifier = Modifier.fillMaxWidth()) {
218              Text(stringResource(R.string.hot_potato_red_team), style = MaterialTheme.typography.titleSmall) // Use stringResource
219              redTeam.forEach { player -> Text(" - ${player.name}") }
220              Spacer(modifier = Modifier.height(8.dp))
221              Text(stringResource(R.string.hot_potato_blue_team), style = MaterialTheme.typography.titleSmall) // Use stringResource
222              blueTeam.forEach { player -> Text(" - ${player.name}") }
223          }
224      }
